//API Login

test('Login via API and reuse token', async ({ request, page }) => {

  const response = await request.post('https://example.com/api/login', {
    data: {
      username: 'testuser',
      password: 'password123'
    }
  });

  const body = await response.json();
  const token = body.token;

  // Set token in browser
  await page.addInitScript(token => {
    localStorage.setItem('authToken', token);
  }, token);

  await page.goto('https://example.com/dashboard');

  await expect(page).toHaveURL(/dashboard/);
});

await page.waitForSelector('#loginBtn', { state: 'visible' });

test('Multiple users', async ({ page }) => {
  const users = [
    { user: 'u1', pass: 'p1' },
    { user: 'u2', pass: 'p2' }
  ];

  for (const u of users) {
    await page.goto('https://example.com/login');
    await page.fill('#username', u.user);
    await page.fill('#password', u.pass);
    await page.click('#loginBtn');
  }
});

const { test, expect } = require('@playwright/test');

test('E2E Login Validation', async ({ page }) => {

    await page.goto('https://example.com/login');

    await page.fill('#username', 'testuser');
    await page.fill('#password', 'password123');

    await Promise.all([
        page.waitForNavigation(),
        page.click('#loginBtn')
    ]);

    // Validate login success
    await expect(page.locator('.welcome-user')).toHaveText(/testuser/i);
});
